   347	function PreviewBody(props: {
   348	  state: Extract<PreviewState, { status: "ready" }>;
   349	  onOpenSystem: () => void;
   350	}) {
   351	  const { state } = props;
   352	  switch (state.info.kind) {
   353	    case "markdown":
   354	      return <MarkdownPreview text={state.text ?? ""} />;
   355	    case "json":
   356	    case "code":
   357	    case "text":
   358	      return <CodePreview text={state.text ?? ""} extension={state.info.extension} />;
   359	    case "html":
   360	      return state.fileUrl ? <HtmlPreview url={state.fileUrl} name={state.info.name} /> : <MissingPreview />;
   361	    case "image":
   362	      return state.data ? (
   363	        <ImagePreview data={state.data} name={state.info.name} />
   364	      ) : state.fileUrl ? (
   365	        <ImagePreview url={state.fileUrl} name={state.info.name} />
   366	      ) : (
   367	        <MissingPreview />
   368	      );
   369	    case "audio":
   370	      return state.fileUrl ? <AudioPreview url={state.fileUrl} /> : <MissingPreview />;
   371	    case "video":
   372	      return state.fileUrl ? <VideoPreview url={state.fileUrl} /> : <MissingPreview />;
   373	    case "pdf":
   374	      return state.data ? <PdfPreview data={state.data} /> : <MissingPreview />;
   375	    case "docx":
   376	      return state.data ? (
   377	        <DocxPreview data={state.data} />
   378	      ) : (
   379	        <ThumbnailFallback state={state} onOpenSystem={props.onOpenSystem} />
   380	      );
   381	    case "spreadsheet":
   382	      return state.data ? <SpreadsheetPreview data={state.data} /> : <MissingPreview />;
   383	    case "presentation":
   384	      return state.data ? (
   385	        <PptxPreview data={state.data} name={state.info.name} onOpenSystem={props.onOpenSystem} />
   386	      ) : (
   387	        <ThumbnailFallback state={state} onOpenSystem={props.onOpenSystem} />
   388	      );
   389	    case "unsupported":
   390	      return <ThumbnailFallback state={state} onOpenSystem={props.onOpenSystem} />;
   391	  }
   392	}

   437	function CodePreview({ text, extension }: { text: string; extension: string }) {
   438	  const settings = useAppStore((state) => state.codePreviewSettings);
   439	  const [wrapOverride, setWrapOverride] = useState<boolean | undefined>();
   440	  const [copied, setCopied] = useState(false);
   441	  const copyTimerRef = useRef<ReturnType<typeof setTimeout> | null>(null);
   442	  const displayText = useMemo(() => normalizeCodePreviewText(text), [text]);
   443	  const plainLines = useMemo(() => splitCodePreviewLines(displayText), [displayText]);
   444	  const highlight = useShikiHighlight(displayText, extension, settings, "FilePreviewPanel");
   445	  const language = highlight.language;
   446	  const wrap = wrapOverride ?? settings.wrapLongLines;
   447	
   448	  useEffect(
   449	    () => () => {
   450	      if (copyTimerRef.current) clearTimeout(copyTimerRef.current);
   451	    },
   452	    []
   453	  );
   454	
   455	  useEffect(() => {
   456	    setWrapOverride(undefined);
   457	  }, [settings.wrapLongLines]);
   458	
   459	  const handleCopy = async () => {
   460	    try {
   461	      await navigator.clipboard.writeText(text);
   462	      setCopied(true);
   463	      if (copyTimerRef.current) clearTimeout(copyTimerRef.current);
   464	      copyTimerRef.current = setTimeout(() => setCopied(false), 2000);
   465	    } catch (err) {
   466	      console.error("[FilePreviewPanel] 复制文件内容失败", { err });
   467	    }
   468	  };
   469	
   470	  return (
   471	    <div
   472	      className="relative h-full min-h-0 bg-canvas"
   473	      data-code-font-size={settings.fontSize}
   474	      data-code-line-numbers="true"
   475	      data-language={language}
   476	      data-testid="file-code-preview"
   477	      style={codePreviewInlineStyle(settings)}
   478	    >
   479	      <div className="absolute right-3 top-2 z-10 flex items-center gap-1 rounded-sm bg-canvas/85 p-0.5 backdrop-blur">
   480	        <TooltipProvider delayDuration={300}>
   481	          <Tooltip>
   482	            <TooltipTrigger asChild>
   483	              <button
   484	                type="button"
   485	                className={ICON_BUTTON}
   486	                aria-label={wrap ? "关闭自动换行" : "自动换行"}
   487	                aria-pressed={wrap}
   488	                onClick={() => setWrapOverride((value) => !(value ?? settings.wrapLongLines))}
   489	              >
   490	                <UndoIcon className="size-3.5" />
   491	              </button>
   492	            </TooltipTrigger>
   493	            <TooltipContent>{wrap ? "关闭自动换行" : "自动换行"}</TooltipContent>
   494	          </Tooltip>
   495	          <Tooltip>
   496	            <TooltipTrigger asChild>
   497	              <button
   498	                type="button"
   499	                className={ICON_BUTTON}
   500	                aria-label="复制文件内容"
   501	                onClick={() => void handleCopy()}
   502	              >
   503	                {copied ? (
   504	                  <CheckMediumIcon className="size-3.5" />
   505	                ) : (
   506	                  <CopyIcon className="size-3.5" />
   507	                )}
   508	              </button>
   509	            </TooltipTrigger>
   510	            <TooltipContent>{copied ? "已复制" : "复制文件内容"}</TooltipContent>
   511	          </Tooltip>
   512	        </TooltipProvider>
   513	      </div>
   514	      <div
   515	        className={cn("h-full overflow-auto py-3 pl-4 pr-20 font-mono text-[var(--cxb-code-font-size,12px)] leading-[var(--cxb-code-line-height,20px)]", wrap && "overflow-x-hidden")}
   516	        data-code-wrap={wrap ? "true" : "false"}
   517	      >
   518	        <pre className={cn("m-0", wrap && "whitespace-pre-wrap break-all")}>

   563	function ImagePreview({ data, url, name }: { data?: ArrayBuffer; url?: string; name: string }) {
   564	  const [objectUrl, setObjectUrl] = useState<string>();
   565	
   566	  useEffect(() => {
   567	    if (!data) {
   568	      setObjectUrl(undefined);
   569	      return;
   570	    }
   571	    if (typeof URL.createObjectURL !== "function") {
   572	      console.warn("[FilePreviewPanel] 图片 Blob URL 能力不可用", { name });
   573	      setObjectUrl(undefined);
   574	      return;
   575	    }
   576	    const nextUrl = URL.createObjectURL(new Blob([data], { type: imageMimeTypeForPath(name) }));
   577	    setObjectUrl(nextUrl);
   578	    return () => {
   579	      URL.revokeObjectURL(nextUrl);
   580	    };
   581	  }, [data, name]);
   582	
   583	  const src = objectUrl ?? url;
   584	  if (!src) {
   585	    return <MissingPreview />;
   586	  }
   587	
   588	  return (
   589	    <div className="flex h-full items-center justify-center overflow-auto bg-canvas-soft-2 p-4">
   590	      <img src={src} alt={name} className="max-h-full max-w-full object-contain shadow-sm" />
   591	    </div>
   592	  );
   593	}

   595	function imageMimeTypeForPath(path: string): string {
   596	  const extension = path.split(".").pop()?.toLowerCase();
   597	  switch (extension) {
   598	    case "jpg":
   599	    case "jpeg":
   600	      return "image/jpeg";
   601	    case "gif":
   602	      return "image/gif";
   603	    case "webp":
   604	      return "image/webp";
   605	    case "avif":
   606	      return "image/avif";
   607	    case "bmp":
   608	      return "image/bmp";
   609	    case "svg":
   610	      return "image/svg+xml";
   611	    case "png":
   612	    default:
   613	      return "image/png";
   614	  }

   642	function PdfPreview({ data }: { data: ArrayBuffer }) {
   643	  const { t } = useTranslation();
   644	  const canvasRef = useRef<HTMLCanvasElement>(null);
   645	  const [state, setState] = useState<{
   646	    status: "loading" | "ready" | "error";
   647	    error?: string;
   648	    page: number;
   649	    pageCount: number;
   650	    scale: number;
   651	  }>({ status: "loading", page: 1, pageCount: 0, scale: 1.15 });
   652	
   653	  useEffect(() => {
   654	    let cancelled = false;
   655	    let cleanup: (() => void) | undefined;
   656	    void (async () => {
   657	      try {
   658	        const pdfjs = await import("pdfjs-dist");
   659	        pdfjs.GlobalWorkerOptions.workerSrc = pdfWorkerUrl;
   660	        const task = pdfjs.getDocument({ data: pdfData(data) });
   661	        const pdf = await task.promise;
   662	        if (cancelled) {
   663	          await pdf.cleanup();
   664	          return;
   665	        }
   666	        setState((current) => ({ ...current, status: "ready", pageCount: pdf.numPages }));
   667	        cleanup = () => {
   668	          void pdf.cleanup();
   669	        };
   670	      } catch (error) {
   671	        if (!cancelled) {
   672	          console.warn("[FilePreviewPanel] PDF 加载失败", error);
   673	          setState((current) => ({
   674	            ...current,
   675	            status: "error",
   676	            error: error instanceof Error ? error.message : String(error)
   677	          }));
   678	        }
   679	      }
   680	    })();
   681	    return () => {
   682	      cancelled = true;
   683	      cleanup?.();
   684	    };
   685	  }, [data]);
   686	
   687	  useEffect(() => {
   688	    if (state.status !== "ready" || !canvasRef.current) {
   689	      return;
   690	    }
   691	    let cancelled = false;
   692	    let renderTask: { cancel(): void; promise: Promise<unknown> } | undefined;
   693	    void (async () => {
   694	      try {
   695	        const pdfjs = await import("pdfjs-dist");
   696	        pdfjs.GlobalWorkerOptions.workerSrc = pdfWorkerUrl;
   697	        const pdf = await pdfjs.getDocument({ data: pdfData(data) }).promise;
   698	        const page = await pdf.getPage(state.page);
   699	        const viewport = page.getViewport({ scale: state.scale });
   700	        const canvas = canvasRef.current;
   701	        if (!canvas || cancelled) {
   702	          await pdf.cleanup();
   703	          return;
   704	        }
   705	        const outputScale = window.devicePixelRatio || 1;
   706	        canvas.width = Math.floor(viewport.width * outputScale);
   707	        canvas.height = Math.floor(viewport.height * outputScale);
   708	        canvas.style.width = `${Math.floor(viewport.width)}px`;
   709	        canvas.style.height = `${Math.floor(viewport.height)}px`;
   710	        renderTask = page.render({
   711	          canvas,
   712	          viewport,
   713	          transform: outputScale !== 1 ? [outputScale, 0, 0, outputScale, 0, 0] : undefined
   714	        });
   715	        await renderTask.promise;
   716	        await pdf.cleanup();
   717	      } catch (error) {
   718	        if (!cancelled && !(error instanceof Error && error.name === "RenderingCancelledException")) {
   719	          console.warn("[FilePreviewPanel] PDF 页面渲染失败", error);
   720	          setState((current) => ({
   721	            ...current,
   722	            status: "error",
   723	            error: error instanceof Error ? error.message : String(error)
   724	          }));
   725	        }
   726	      }
   727	    })();
   728	    return () => {
   729	      cancelled = true;
   730	      renderTask?.cancel();
   731	    };
   732	  }, [data, state.page, state.scale, state.status]);
   733	
   734	  if (state.status === "loading") {
   735	    return <CenteredLoader />;
   736	  }
   737	  if (state.status === "error") {
   738	    return <InlineError message={`${t("rightPanel.fileLoadFailed")}：${state.error ?? ""}`} />;
   739	  }
   740	  return (
   741	    <div className="flex h-full min-h-0 flex-col">
   742	      <div className="flex flex-none items-center justify-center gap-1 border-b px-3 py-2">
   743	        <button
   744	          type="button"
   745	          title={t("rightPanel.prevPage")}
   746	          disabled={state.page <= 1}
   747	          onClick={() => setState((current) => ({ ...current, page: Math.max(1, current.page - 1) }))}
   748	          className={ICON_BUTTON}
   749	        >
   750	          <ArrowLeftIcon className="size-3.5" />
   751	        </button>
   752	        <span className="px-2 font-mono text-micro text-muted-foreground">
   753	          {state.page} / {state.pageCount}
   754	        </span>
   755	        <button
   756	          type="button"
   757	          title={t("rightPanel.nextPage")}
   758	          disabled={state.page >= state.pageCount}
   759	          onClick={() =>
   760	            setState((current) => ({ ...current, page: Math.min(current.pageCount, current.page + 1) }))
   761	          }
   762	          className={ICON_BUTTON}
   763	        >
   764	          <ArrowLeftIcon className="size-3.5 rotate-180" />
   765	        </button>
   766	        <span className="mx-1 h-4 w-px bg-border" />
   767	        <button
   768	          type="button"
   769	          title={t("rightPanel.zoomOut")}
   770	          onClick={() => setState((current) => ({ ...current, scale: Math.max(0.6, current.scale - 0.15) }))}
   771	          className={ICON_BUTTON}
   772	        >
   773	          <MinusIcon className="size-3.5" />
   774	        </button>
   775	        <button
   776	          type="button"
   777	          title={t("rightPanel.zoomIn")}
   778	          onClick={() => setState((current) => ({ ...current, scale: Math.min(2.4, current.scale + 0.15) }))}
   779	          className={ICON_BUTTON}
   780	        >
   781	          <PlusIcon className="size-3.5" />
   782	        </button>
   783	      </div>
   784	      <div className="min-h-0 flex-1 overflow-auto bg-canvas-soft-2 p-4">
   785	        <canvas ref={canvasRef} className="mx-auto bg-white shadow-sm" />
   786	      </div>
   787	    </div>
   788	  );
   789	}

   932	function PptxPreview({
   933	  data,
   934	  name,
   935	  onOpenSystem
   936	}: {
   937	  data: ArrayBuffer;
   938	  name: string;
   939	  onOpenSystem: () => void;
   940	}) {
   941	  const { t } = useTranslation();
   942	  const scrollRef = useRef<HTMLDivElement>(null);
   943	  const containerRef = useRef<HTMLDivElement>(null);
   944	  const viewerRef = useRef<PptxViewerInstance | null>(null);
   945	  const [state, setState] = useState<{
   946	    status: "loading" | "ready" | "error";
   947	    error?: string;
   948	    page: number;
   949	    pageCount: number;
   950	    zoom: number;
   951	    busy: boolean;
   952	  }>({ status: "loading", page: 1, pageCount: 0, zoom: 100, busy: false });
   953	
   954	  useEffect(() => {
   955	    const container = containerRef.current;
   956	    const scrollContainer = scrollRef.current;
   957	    if (!container || !scrollContainer) {
   958	      return;
   959	    }
   960	    const abortController = new AbortController();
   961	    let cancelled = false;
   962	    container.innerHTML = "";
   963	    viewerRef.current = null;
   964	    setState({ status: "loading", page: 1, pageCount: 0, zoom: 100, busy: false });
   965	
   966	    void (async () => {
   967	      try {
   968	        console.info("[FilePreviewPanel] 开始渲染 PPTX 预览", { name, bytes: data.byteLength });
   969	        const { PptxViewer, RECOMMENDED_ZIP_LIMITS } = await import("@aiden0z/pptx-renderer");
   970	        const viewer = await PptxViewer.open(data.slice(0), container, {
   971	          fitMode: "contain",
   972	          zoomPercent: 100,
   973	          scrollContainer,
   974	          zipLimits: RECOMMENDED_ZIP_LIMITS,
   975	          pdfjs: {
   976	            moduleUrl: pdfjsModuleUrl,
   977	            workerUrl: pdfjsWorkerUrl
   978	          },
   979	          listOptions: {
   980	            windowed: true,
   981	            batchSize: 8,
   982	            initialSlides: 4,
   983	            overscanViewport: 1.5,
   984	            showSlideLabels: true
   985	          },
   986	          signal: abortController.signal,
   987	          onSlideChange: (index) => {
   988	            setState((current) => ({ ...current, page: index + 1 }));
   989	          },
   990	          onSlideError: (index, error) => {
   991	            console.warn("[FilePreviewPanel] PPTX 单页渲染失败", { name, page: index + 1, error });
   992	          },
   993	          onNodeError: (nodeId, error) => {
   994	            console.warn("[FilePreviewPanel] PPTX 节点渲染失败", { name, nodeId, error });
   995	          }
   996	        });
   997	        if (cancelled) {
   998	          viewer.destroy();
   999	          return;
  1000	        }
  1001	        viewerRef.current = viewer;
  1002	        console.info("[FilePreviewPanel] PPTX 预览渲染完成", {
  1003	          name,
  1004	          slideCount: viewer.slideCount
  1005	        });
  1006	        setState({
  1007	          status: "ready",
  1008	          page: Math.max(1, viewer.currentSlideIndex + 1),
  1009	          pageCount: viewer.slideCount,
  1010	          zoom: viewer.zoomPercent,
  1011	          busy: false
  1012	        });
  1013	      } catch (error) {
  1014	        if (!cancelled && !abortController.signal.aborted) {
  1015	          console.warn("[FilePreviewPanel] PPTX 预览渲染失败", { name, error });
  1016	          setState({
  1017	            status: "error",
  1018	            page: 1,
  1019	            pageCount: 0,
  1020	            zoom: 100,
  1021	            busy: false,
  1022	            error: error instanceof Error ? error.message : String(error)
  1023	          });
  1024	        }
  1025	      }
  1026	    })();
  1027	
  1028	    return () => {
  1029	      cancelled = true;
  1030	      abortController.abort();
  1031	      viewerRef.current?.destroy();
  1032	      viewerRef.current = null;
  1033	      container.innerHTML = "";
  1034	    };
  1035	  }, [data, name]);

  1037	  async function goToSlide(page: number): Promise<void> {
  1038	    const viewer = viewerRef.current;
  1039	    if (!viewer) {
  1040	      return;
  1041	    }
  1042	    const targetPage = Math.min(Math.max(1, page), Math.max(1, state.pageCount));
  1043	    setState((current) => ({ ...current, busy: true }));
  1044	    try {
  1045	      await viewer.goToSlide(targetPage - 1, { behavior: "smooth", block: "start" });
  1046	      setState((current) => ({ ...current, page: targetPage, busy: false }));
  1047	    } catch (error) {
  1048	      console.warn("[FilePreviewPanel] PPTX 跳转页面失败", { name, page: targetPage, error });
  1049	      setState((current) => ({ ...current, busy: false }));
  1050	    }
  1051	  }

  1053	  async function setZoom(zoom: number): Promise<void> {
  1054	    const viewer = viewerRef.current;
  1055	    if (!viewer) {
  1056	      return;
  1057	    }
  1058	    const nextZoom = Math.min(200, Math.max(50, zoom));
  1059	    setState((current) => ({ ...current, busy: true }));
  1060	    try {
  1061	      await viewer.setZoom(nextZoom);
  1062	      setState((current) => ({ ...current, zoom: viewer.zoomPercent, busy: false }));
  1063	    } catch (error) {
  1064	      console.warn("[FilePreviewPanel] PPTX 缩放失败", { name, zoom: nextZoom, error });
  1065	      setState((current) => ({ ...current, busy: false }));
  1066	    }
  1067	  }
  1068	
  1069	  if (state.status === "error") {
  1070	    return (
  1071	      <div className="flex h-full flex-col items-center justify-center gap-3 px-6 text-center text-caption text-muted-foreground">
  1072	        <WarningCircleIcon className="size-5 text-warning" />
  1073	        <div className="max-w-[360px] space-y-1">
  1074	          <p className="text-foreground">PPTX 内嵌预览失败</p>
  1075	          <p>{`${t("rightPanel.fileLoadFailed")}：${state.error ?? ""}`}</p>
  1076	        </div>
  1077	        <button
  1078	          type="button"
  1079	          onClick={onOpenSystem}
  1080	          className="rounded-sm border bg-card px-3 py-1.5 text-micro text-foreground transition-colors hover:bg-muted"
  1081	        >
  1082	          {t("rightPanel.openWithSystem")}
  1083	        </button>
  1084	      </div>
  1085	    );
  1086	  }

  1243	function formatBytes(bytes: number): string {
  1244	  if (bytes < 1024) return `${bytes} B`;
  1245	  const kb = bytes / 1024;
  1246	  if (kb < 1024) return `${kb.toFixed(kb >= 10 ? 0 : 1)} KB`;
  1247	  const mb = kb / 1024;
  1248	  if (mb < 1024) return `${mb.toFixed(mb >= 10 ? 0 : 1)} MB`;
  1249	  return `${(mb / 1024).toFixed(1)} GB`;
